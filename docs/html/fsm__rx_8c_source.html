114static void do_store_data(fsm_t *p_this)
115{
116 fsm_rx_t *p_fsm = (fsm_rx_t *)(p_this);
117 uint16_t *buffer_edges = port_rx_get_buffer_edges(p_fsm->rx_id);
118 uint32_t num_edges = port_rx_get_num_edges(p_fsm->rx_id);
119 uint32_t *p_code = (uint32_t *)(&(p_fsm->code));
120 p_fsm->is_repetition = fsm_rx_NEC_parse_code(p_fsm->p_fsm_rx_nec, buffer_edges, num_edges, p_code);
121 p_fsm->is_error = (p_fsm->code == 0x00) && (!p_fsm->is_repetition);
122 p_fsm->num_edges_detected = 0;
123 port_rx_clean_buffer(p_fsm->rx_id);
124}
