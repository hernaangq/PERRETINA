47static void _timer_symbol_setup()
48{
49 /* Primero , habilitamos siempre el reloj del timer */
50 RCC -> APB2ENR |= RCC_APB2ENR_TIM1EN ; /* TIM1 con APB2 . */
51
52 /* Temporizacion */
53 /* 1) Fuente de reloj : interno ( bits CKD =00) */
54 TIM1 -> CR1 = 0x0000 ; /* Aunque es el valor por defecto */
55
56 /* 2) Deshabilita el contador (no es indispensable). */
57 TIM1 -> CR1 &= ~ TIM_CR1_CEN ;
58
59 /* Habilita el preload (no es indispensable). */
60 TIM1->CR1 |= TIM_CR1_ARPE;
61
62 /* 3) Aseguramos inicio del contador a 0 */
63 TIM1->CNT = 0;
64
65 //CALCULOS
66 uint32_t psmin = (((SystemCoreClock * NEC_TX_TIMER_TICK_BASE_US /1000000))/(65536)) - 1 ;
67 psmin = 0;
68 uint32_t arrmin = (((SystemCoreClock * NEC_TX_TIMER_TICK_BASE_US / 1000000))/(psmin + 1)) - 1;
69
70 /* 4) Cargamos el autorreload */
71 TIM1->ARR = arrmin;
72
73 /* 5) Cargamos el prescaler */
74 TIM1->PSC = psmin;
75
76 /* 6) ( IMPORTANTE ) Re - inicializa el contador y actualiza los registros .
77 IMPORTANTE realizarlo despues de cargar ARR y PSC */
78 /* Los valores de PSC y ARR estan en los registro de precarga . Para cargarlos
79 en los registros activos , necesitamos que se actualice un evento . Esto
80 podemos hacerlo manualmente actualizando el registro TIMx - >EGR , o esperar a
81 que pase el timer . */
82 TIM1->EGR = TIM_EGR_UG; /* Genera evento de actualizacion */
83
84 /* Interrupciones */
85 /* 7) Limpiamos las interrupciones (si hubiese ) */
86 TIM1->SR = ~TIM_SR_UIF;
87
88 /* 8) Habilitamos las interrupciones en el registro */
89 TIM1->DIER |= TIM_DIER_UIE; /* Interrumpe al acabar de contar */
90
91 /* Finally, set the priority and enable interruptions globally. */
92 NVIC_SetPriority(TIM1_UP_TIM10_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), 1, 0)); /* Priority 1, sub-priority 0 */
93 NVIC_EnableIRQ(TIM1_UP_TIM10_IRQn); /* Enable interrupt */
94}
