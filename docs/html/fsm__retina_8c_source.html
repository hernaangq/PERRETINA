214static bool check_short_pressed(fsm_t *p_this)
215{
216 fsm_retina_t *p_fsm_retina = (fsm_retina_t *)(p_this); // cast p_this
217 int cond1 = (fsm_button_get_duration(p_fsm_retina->p_fsm_button) != 0);
218 int cond2 = (fsm_button_get_duration(p_fsm_retina->p_fsm_button) < p_fsm_retina->long_button_press_ms);
219 return cond1 && cond2;
220}

226static bool check_long_pressed(fsm_t *p_this)
227{
228 fsm_retina_t *p_fsm_retina = (fsm_retina_t *)(p_this); // cast p_this
229 int cond1 = (fsm_button_get_duration(p_fsm_retina->p_fsm_button) != 0);
230 int cond2 = (fsm_button_get_duration(p_fsm_retina->p_fsm_button) >= p_fsm_retina->long_button_press_ms);
231 return cond1 && cond2;

278static bool check_activity(fsm_t *p_this)
279{
280 fsm_retina_t *p_fsm_retina = (fsm_retina_t *)(p_this);
281 bool cond1 = fsm_button_check_activity(p_fsm_retina->p_fsm_button);
282 bool cond2 = fsm_tx_check_activity(p_fsm_retina->p_fsm_tx);
283 bool cond3 = fsm_rx_check_activity(p_fsm_retina->p_fsm_rx);
284 bool cond4 = fsm_sensor_check_activity(p_fsm_retina->p_fsm_sensor);
285 return (cond1 | cond2 | cond3 | cond4);
286}
