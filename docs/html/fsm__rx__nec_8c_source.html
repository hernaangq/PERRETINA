294bool fsm_rx_NEC_parse_code(fsm_t *p_this,
295 uint16_t *p_edge_ticks,
296 uint32_t num_edges,
297 uint32_t *p_code)
298{
299 fsm_rx_nec_t *p_fsm = (fsm_rx_nec_t *)(p_this);
300 p_fsm->f.current_state = NEC_IDLE;
301 p_fsm->code = 0;
302 p_fsm->is_repetition = false;
303 p_fsm->p_edge_ticks = p_edge_ticks;
304 p_fsm->num_edges_to_read = num_edges;
305 while (p_fsm->num_edges_to_read > 1)
306 {
307 fsm_fire(&p_fsm->f);
308 }
309 *p_code = p_fsm->code;
310
311 return p_fsm->is_repetition;
312}
